### Your solution below:
file_list = [
    "data_quiz_1/china_co2_emissions.csv",
    "data_quiz_1/germany_co2_emissions.csv",
]

list_emissions = []

for filename in file_list:                          
    df = pd.read_csv(filename)                      
    mean_value = df["total_emissions"].mean()    
    list_emissions.append(mean_value)   
    print("File processed:", filename)
    print("Mean of total_emissions:", mean_value)
    
print("Final list of average total_emissions:")
print(list_emissions)
